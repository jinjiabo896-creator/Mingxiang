<script>
    // 维度 5：动态音频库（以后你只需要在这里增加数据）
    const audioLibrary = {
        sleep: [
            {title: "林间细雨", url: "https://actions.google.com/sounds/v1/water/rain_on_roof.ogg"},
            {title: "深海波动", url: "https://actions.google.com/sounds/v1/water/waves_crashing_on_rock.ogg"}
        ],
        focus: [
            {title: "白噪音工作坊", url: "https://www.soundhelix.com/examples/mp3/SoundHelix-Song-2.mp3"}
        ]
        // 更多音频...
    };

    let isPlaying = false;
    const audio = document.getElementById('audio-engine');

    function togglePlayer(show) {
        document.getElementById('player').classList.toggle('active', show);
    }

    function playAudio(title, url) {
        document.getElementById('current-title').innerText = title;
        audio.src = url;
        audio.play();
        isPlaying = true;
        document.getElementById('play-trigger').innerText = "⏸";
        togglePlayer(true);
    }

    function toggleAudio() {
        if(isPlaying) {
            audio.pause();
            document.getElementById('play-trigger').innerText = "▶";
        } else {
            audio.play();
            document.getElementById('play-trigger').innerText = "⏸";
        }
        isPlaying = !isPlaying;
    }

    function loadCategory(id) {
        const item = audioLibrary[id] ? audioLibrary[id][0] : null;
        if(item) {
            playAudio(item.title, item.url);
            document.getElementById('current-cat').innerText = "分类推演播放";
        } else {
            alert("该类目新音频正在录制中，敬请期待...");
        }
    }

    // 初始化：根据时间更新欢迎语
    const hour = new Date().getHours();
    const greet = hour < 12 ? "早上好" : (hour < 18 ? "下午好" : "晚上好");
    document.getElementById('greeting').innerText = `${greet}，修行者`;
</script>
